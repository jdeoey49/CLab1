bool isEven(int num)// isEven function for problem 1
{
	if (num % 2 == 0)
		return 1;
	else
		return 0;
}
bool multiple(int num1, int num2)//multiple function for problem 2
{
	if (num1 == 0)
		return 0;
	else if (num2 % num1 != 0)
		return 0;
	else
		return 1;
}
double hypotenuse(double num1, double num2)//hypotenuse function for problem 3
{
	double ans;
	ans = sqrt(num1*num1 + num2 * num2);
	return ans;
}

int celsius(int temperatureC)//celsius function for problem5
{
	int temperatureF;
	temperatureF = temperatureC * 9 / 5 + 32;
	return temperatureF;
}
int Fahrenheit(int temperatureF)//farenheit function for problem5
{
	int temperatureC;
	temperatureC = (temperatureF-32)*5/9 ;
	return temperatureC;
}
