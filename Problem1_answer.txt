void forthTask() {
	//dont forget to put #define PI 3.14159 before the function 
	//we need three variables now we call them a, b and c 
	//and they are all of the type integer
	int a, b, c;
	//print some stuff on the command line
	std::cout << "THIS IS TASK 4" << std::endl;
	//ask the user for 3 integer numbers and store them into a,b,c
	// if you want to type the first value in the same line remove the std:endl here
	std::cout << "Please enter three integer numbers : " << std::endl;
	std::cin >> a;
	std::cin >> b;
	std::cin >> c;
	//compute the sum 
	int sum = a + b + c;
	//now its time to find out which one is highest and which one is the lowest
	//poshi and poslow is only for a future task and not necessary here
	int poshi, poslow;
	//we will use two variables to store the values for the neccessary comparison process
	//those values are
	//highest which will store the highest value
	//and lowest which will store the lowest value 
	int highest, lowest;
	if (a > b) {
		//if a is bigger than b,
		//a is currently the highest number and b the lowest
		poshi = 1;
		poslow = 2;
		highest = a;
		lowest = b;
	}
	else{
		//if b is bigger than a,
		//b is currently the highest number and c the lowest
		//if a is equal to b,
		//a is currently the lowest number
		//while at the same time the lowest number is the highest number
		poshi = 2;
		poslow = 1;
		highest = b;
		lowest = a;
	}
	// if c is higher than the highest number from previous comparison 
	// the variable value of highest becomes the value of c
	if (c >= highest) {		
		highest = c;
		poshi = 3;
	}
	// if c is lower than the lowest number from previous comparison
	//the variable value of lowest becomes the value of c
	if (c <= lowest) {		
		lowest = c;
		poslow = 3;
	}
	//print out the results
	std::cout << "The sum of those three number is : " << sum << std::endl;
	std::cout << "The average of those three number is : " << sum / 3.0 << std::endl;
	std::cout << "The product of those three number is : " << a * b * c << std::endl;
	std::cout << "The lowest of those three number is : " << lowest << std::endl;
	std::cout << "The highest of those three number is : " << highest << std::endl;
	//Finished Forth Task
}
